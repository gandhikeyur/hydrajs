# hydraja
Client side API for Hydra

# Tests
Set HYDRA_HOSTNAME, RHN_USER, and RHN_PASS environment variables.

    export HYDRA_HOSTNAME=;export RHN_USER=;export RHN_PASS=
    npm run test

# Contributing

    [remote "upstream"]

	fetch = +refs/heads/*:refs/remotes/upstream/*
